  • вначале числа кратные двум, от наибольшего к наименьшему;
  • затем оставшиеся числа от наименьшего к наибольшему.
let input = [-2, 2, 4, 6, 8, 10, 3, 5, 7, 9, -1, -11];
func(input); // [ 10, 8, 6, 4, 2, -2, -11, -1, 3, 5, 7, 9 ]
